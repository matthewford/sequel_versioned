242       # Compares two version strings
243       def version_compare(v1, v2)
244         v1 = v1.split(".").map { |i| i.to_i }
245         v2 = v2.split(".").map { |i| i.to_i }
246         parts = [v1.length, v2.length].max
247         v1.push 0 while v1.length < parts
248         v2.push 0 while v2.length < parts
249         v1.zip(v2).each do |a,b|
250           return -1 if a < b
251           return  1 if a > b
252         end
253         return 0
254       end
